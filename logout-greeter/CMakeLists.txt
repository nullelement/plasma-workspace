    PW::KWorkspace
    Qt5::Widgets
    Qt5::Quick
    Qt5::X11Extras
    Qt5::DBus
    KF5::Declarative
    KF5::I18n
    KF5::Package
    KF5::QuickAddons
    KF5::WaylandClient
    KF5::WindowSystem
    ${X11_LIBRARIES}
)
